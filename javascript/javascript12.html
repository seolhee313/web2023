01. 데이터 이펙트 : 1부터 100까지 출력하기

for문을 이용해서 1-100까지 출력하는 예제입니다.

for(let i=1; i<=100; i++){   //반복문 for문
    console.log(i)  //초기값을 1으로 설정하고 1씩 증가시켜서 100번을 반복함
}

04. 데이터 이펙트 : 1부터 100까지 5의 배수 출력하기

for문을 이용해서 0-100까지 5의 배수만 출력하는 예제입니다. 조건문 if문과 증감연산자를 이용한 방법입니다.

// 1. 증감 연산자를 이용한 방법
for(let i=0; i<=100; i+=5){
    console.log(i)  //초기값을 0으로 설정하고 5씩 증가시켜서 100번을 반복함
};


// 2. if문을 이용한 방법
for(i=0; i<=100; i++){
    if(i % 5 == 0){
        console.log(i)  // % 연산자를 이용하여 나머지가 0이면 짝수를 의미한다.
    }
}

05. 데이터 이펙트 : 1부터 100까지 짝수(파란색) 홀수(빨간색) 출력하기

for문을 이용해서 1부터 100까지 짝수(파란색) 홀수(빨간색) 출력할 수 있다.

for(let i=1; i<=99; i++){
    if( i % 2 ==0){
        document.write("<span style='color:blue'>"+i+"</span>");
    }
    if( i % 2 ==1){
        document.write("<span style='color:red'>"+i+"</span>");
    }                
}

06. 데이터 이펙트 : for문을 이용해서 100칸 테이블 만들기

for문을 이용해서 100칸 테이블을 만들 수 있다.

let table = "<table border='1'>";
    for(let i=1; i<=10; i++){
        table += "<tr>";
        for(let j=1; j<=10; j++){
            table += "<td>"+j+"</td>";
        }
        table += "</tr>";
    }
    table += "</table>";
document.write(table);

07. 데이터 이펙트 : 중첩 for문을 이용해서 구구단 만들기

중첩for문을 이용해서 구구단을 만들 수 있다.

for(let i=2; i<=9; i++){
    document.write(i + "단","<br>");

    for(let j=1; j<=9; j++){
        document.write(i + "*" + j + "=" + i*j);
        document.write("<br>")
    }

    document.write("<br><br>");
}
